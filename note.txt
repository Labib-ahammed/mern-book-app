import React, { useState } from "react";

const Navbar = () => {
  const [searchTerm, setSearchTerm] = useState("");

  const handleSearch = async (e) => {
    e.preventDefault();
    if (!searchTerm) {
      return;
    }
    try {
      const response = fetch(
        `http://localhost:5000/api/booksearch?query=${searchTerm}`
      );
      const data = await response.json();
    } catch (error) {
      console.error("Error fetching search results:", error);
    }
  };
  return (
    <nav className="w-full flex justify-center items-center p-3">
      <form onSubmit={handleSearch}>
        <input
          placeholder="Enter book name to search..."
          type="search"
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          className="border-2 border-gray-700 w-4/5 p-2 rounded-xl md:w-1/3"
        />
        <button
          type="submit"
          className="p-2 bg-gray-500 text-white border rounded-md"
        >
          Search
        </button>
      </form>
    </nav>
  );
};

export default Navbar;
